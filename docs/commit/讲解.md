```rust
pub fn first_of_sequence(symbols: &[Symbol], first_sets: &FirstSets) -> (BTreeSet<String>, bool) {
    // 第 1 句：如果符号串本身为空，就认为它可空，但 FIRST 集里没有实际终结符
    if symbols.is_empty() {
        return (BTreeSet::new(), true);
    }

    // 第 2 句：first 保存不含 ε 的 FIRST 结果
    let mut first = BTreeSet::new();

    // 第 3 句：nullable 用来记录这个符号串是否可空
    let mut nullable = true;

    for (index, symbol) in symbols.iter().enumerate() {
        match symbol {
            Symbol::Terminal(value) => {
                // 第 4 句：遇到终结符，FIRST 直接加入它本身，并停止继续向后看
                first.insert(value.clone());
                nullable = false;
                break;
            }
            Symbol::NonTerminal(name) => {
                // 第 5 句：遇到非终结符，就把它已有的 FIRST 展开进来
                let nested_first = first_sets
                    .get(name)
                    .unwrap_or_else(|| panic!("missing FIRST set for non-terminal: {name}"));
                for entry in nested_first {
                    if entry != "ε" {
                        first.insert(entry.clone());
                    }
                }
                // 第 6 句：如果当前非终结符不可空，就在这里停止
                if !nested_first.contains("ε") {
                    nullable = false;
                    break;
                }
                // 第 7 句：如果它可空，就继续看后面的符号
            }
            Symbol::Epsilon => {
                // 第 8 句：这里约束 ε 只能单独作为一个符号串出现
                assert!(
                    symbols.len() == 1 && index == 0,
                    "epsilon must appear alone in a sequence"
                );
                break;
            }
            Symbol::EndMarker => {
                // 第 9 句：如果遇到结束符，就把 $ 视为当前可见终结符
                first.insert("$".to_string());
                nullable = false;
                break;
            }
        }
    }

    // 第 10 句：最终返回“不含 ε 的 FIRST 集”和“这个串是否可空”
    (first, nullable)
}
```

我会这样讲：

“这段代码是我这一部分最核心的地方。它做的事情其实很明确，就是计算一个符号串的 `FIRST`。如果当前遇到终结符，就直接把它放进去并结束；如果遇到非终结符，就把那个非终结符已有的 `FIRST` 展开进来；如果那个非终结符不可空，就在这里停下；如果它可空，才继续往后看。”

“这里最关键的设计不是 if-else 本身，而是这个函数最后统一返回两样东西：一个是不含 `ε` 的集合，一个是 `nullable` 标记。这样我后面不管是算 `FIRST`、算 `FOLLOW`，还是构造分析表，都可以复用这套逻辑。”

```rust
pub fn build_parse_table(
    grammar: &Grammar,
    first_sets: &FirstSets,
    follow_sets: &FollowSets,
) -> Result<ParseTable, ParseTableError> {
    // 第 1 句：先检查所有非终结符是否都有 FIRST 集
    for non_terminal in &grammar.non_terminals {
        if !first_sets.contains_key(non_terminal) {
            return Err(ParseTableError::MissingFirstSet {
                non_terminal: non_terminal.clone(),
            });
        }
    }

    // 第 2 句：初始化空的预测分析表
    let mut table = ParseTable {
        entries: BTreeMap::new(),
    };

    for production in &grammar.productions {
        // 第 3 句：先求产生式右部的 FIRST(α) 以及它是否可空
        let (sequence_first, nullable) = first_of_sequence(&production.right, first_sets);

        // 第 4 句：把 FIRST(α) 中的终结符填入 M[A, a]
        for terminal in sequence_first.iter() {
            insert_entry(&mut table, &production.left, terminal, production.id)?;
        }

        if nullable {
            // 第 5 句：如果 α 可空，就继续用 FOLLOW(A) 填表
            let follow = follow_sets.get(&production.left).ok_or_else(|| {
                ParseTableError::MissingFollowSet {
                    non_terminal: production.left.clone(),
                }
            })?;
            for terminal in follow {
                insert_entry(&mut table, &production.left, terminal, production.id)?;
            }
        }
```
